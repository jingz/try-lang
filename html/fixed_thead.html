setTimeout(function(){
    // sync with of the 2 tables
    $('#fixed-head-wrapper table').width($('#grid-result').width());

    // sync each column width of header of 2 tables
    sync_head();
}, 100);

function sync_head() {
    var wid = [];
    $('#grid-result thead tr:first th').each(function(){
        wid.push($(this).width());
    });

    $('#fixed-head-wrapper thead tr:first th').each(function(i){
        $(this).width(wid[i]);
    });
}

$('#constrain-wrapper').scroll(function(){
    var top = ($(this).scrollTop());
    $('#fixed-head-wrapper').css('top', top);
});
